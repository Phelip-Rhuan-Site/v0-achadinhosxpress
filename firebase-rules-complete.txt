rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Função para verificar se o usuário é admin
    function isAdmin() {
      return request.auth != null &&
             exists(/databases/$(database)/documents/invites/$(request.auth.token.email)) &&
             (get(/databases/$(database)/documents/invites/$(request.auth.token.email)).data.role == 'ADM Master' ||
              get(/databases/$(database)/documents/invites/$(request.auth.token.email)).data.role == 'admin');
    }

    // Produtos: leitura pública, escrita só admin
    match /products/{productId} {
      allow read: if true;
      allow write, delete: if isAdmin();
    }

    // Meta (configurações públicas como contato, footer, header): leitura pública, escrita só admin
    match /meta/{document} {
      allow read: if true;
      allow write, delete: if isAdmin();
    }

    // Config (configurações internas): apenas admin
    match /config/{document} {
      allow read, write, delete: if isAdmin();
    }

    // Invites: usuário autenticado pode ler seu próprio convite, admin pode tudo
    match /invites/{email} {
      allow read: if request.auth != null && request.auth.token.email == email;
      allow write, delete: if isAdmin();
      allow create: if isAdmin();
    }

    // Users: usuário pode ler/escrever seus próprios dados, admin pode ler todos
    match /users/{userId} {
      allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
      allow write: if request.auth != null && request.auth.uid == userId;
      allow delete: if isAdmin();
    }

    // Orders: usuário pode ler seus próprios pedidos e criar novos, admin pode tudo
    match /orders/{orderId} {
      allow read: if request.auth != null && 
                     (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if request.auth != null;
      allow update, delete: if isAdmin();
    }

    // Cart: usuário pode gerenciar seu próprio carrinho
    match /cart/{userId} {
      allow read, write, delete: if request.auth != null && request.auth.uid == userId;
    }

    // Categories: leitura pública, escrita só admin
    match /categories/{categoryId} {
      allow read: if true;
      allow write, delete: if isAdmin();
    }

    // Reviews/Avaliações: usuários autenticados podem criar, admin pode moderar
    match /reviews/{reviewId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && 
                               (resource.data.userId == request.auth.uid || isAdmin());
    }

    // Bloquear acesso a qualquer outra coleção não especificada
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
